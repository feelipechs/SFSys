Tabela alunos:
CREATE TABLE alunos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL);

Tabela cursos:
CREATE TABLE cursos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    descricao TEXT );

Tabela inscricoes:
*(Tabela intermediária para o relacionamento muitos-para-muitos):
CREATE TABLE inscricoes (
    aluno_id INT,
    curso_id INT,
    PRIMARY KEY(aluno_id, curso_id),
    FOREIGN KEY (aluno_id) REFERENCES alunos(id),
    FOREIGN KEY (curso_id) REFERENCES cursos(id));

criar aluno: POST http://localhost:3000/alunos { "nome": "Cristiano Ronaldo", "email": "cristiano.ronaldo@exemplo.com" } 
criar curso: POST http://localhost:3000/cursos { "titulo": "Cálculo de Backend", "descricao": "Aprenda a otimizar APIs complexas." } 
listar alunos: GET	http://localhost:3000/alunos
listar cursos: GET	http://localhost:3000/cursos
inscrever aluno em curso: POST	http://localhost:3000/inscricoes { "aluno_id": 7, "curso_id": 1 }
listar cursos de um aluno: GET http://localhost:3000/alunos/1/cursos
listar alunos de um curso: GET	http://localhost:3000/cursos/2/alunos
